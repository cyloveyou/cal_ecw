cmake_minimum_required(VERSION 3.22)
project(main_ecl)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Debug or Release
set(CMAKE_BUILD_TYPE Debug)
set(xlnt_DIR "/usr/local/lib/cmake/xlnt")
find_package(Xlnt REQUIRED)

include_directories(${CMAKE_SOURCE_DIR}/libxl/include_cpp)
link_directories(${CMAKE_SOURCE_DIR}/libxl/lib64)
add_library(lib_ecl SHARED lib_ecl.cpp)


add_executable(main_ecl main_ecl.cpp)
target_link_libraries(main_ecl lib_ecl xl xlnt::xlnt)
